golang连接MYSQL进行增删改查操作

Posted on 2019-10-18

#我的第一篇博客文章

#在这里用一个示例示范golang如何使用database/sql接口对数据库进行增删盖查操作

#代码部分如下

package main

import (
_"github.com/Go-SQL-Drive/MySQL"
"database/sql"
"fmt"
//"time"
)

func main(){
db, err := sql.Open("mysql", "Richar_Chao:Richar_Chao@/test?charset=utf8")  //数据库名及密码（以我自己的为例）
checkErr(err)
// 开始插入数据
stmt, err := db.Prepare("INSERT userinfo SET usename=?, departname=?,created=?")
checkErr(err)

res, err := stmt.Exec("Richar_Chao", "研发部门", "2019-10-19")
checkErr(err)
id, err := res.LastInsertId()
checkErr(err)
fmt.Println(id)
//更新数据
stmt, err := db.Prepare("update userinfo set usename=?, where uid=?")
checkErr(err)

res , err := stmt.Exec("Richar_Chaoupdate", id)
checkErr(err)

affect, err := res.RowsAffected()
checkErr(err)

fmt.Println(affect)

//查询数据
rows, err := db.Query("SELECT * FROM useinfo")
checkErr(err)

for rows.Next(){
    var uid int
    var username string
    var department string
    var created string
    err = rows.Scan(&uid, &username, &department, &created)
    checkErr(err)
    fmt.Println(uid)
    fmt.Println(username)
    fmt.Println(department)
    fmt.Println(created)

}

//删除数据

stmt, err := db.Prepare("delete from userinfo  where uid=?")
checkErr(err)

res, err := stmt.Exec(id)
checkErr(err)

afect, err := res.RowsAffected()
checkErr(err)

fmt.Println(affect)

db.Close()
}

func checkErr(err error){
if err != nil{
panic(err)
}
}

#通过上面的代码，可以看出实际上golang操纵MYSQL数据库还是很方便的，主要是有驱动的支持。
